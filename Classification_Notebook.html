Introduction

In this analysis, we perform a plant disease epidemiological study using machine learning techniques. We use a dataset containing various environmental and soil properties to predict the Level of Disease in plants. The goal is to:

  • Read a provided dataset with from github
  • Preprocess the data, including scaling and partitioning.
  • Train multiple machine learning models for classification.
  • Compare models based on accuracy and select the best ones.
  • Perform hyperparameter tuning on the top models.
  • Evaluate the final models and interpret the results.
  • Visualize the confusion matrix and quantify feature importance.

Data Preprocessing

We preprocess the data by scaling the predictor variables and partitioning the dataset into training and testing sets.

# Exclude 'LevelOfDisease' and 'Severity' from predictors
predictors <- data[, !(names(data) %in% c("LevelOfDisease", "Severity"))]

# Scale the predictor variables
preProcValues <- preProcess(predictors, method = c("center", "scale"))
predictorsScaled <- predict(preProcValues, predictors)

# Combine scaled predictors with the target variable
dataScaled <- cbind(predictorsScaled, LevelOfDisease = data$LevelOfDisease)

# Stratified partitioning of the dataset into training and testing sets (80/20 split)
trainIndex <- createDataPartition(dataScaled$LevelOfDisease, p = 0.8, list = FALSE)
trainData <- dataScaled[trainIndex, ]
testData <- dataScaled[-trainIndex, ]

# Extract accuracy metrics and compute 95% confidence intervals
accuracy_results <- resamps$values
model_names <- names(models)

# Function to compute mean accuracy and 95% confidence intervals
getMeanCI <- function(x) {
  n <- length(x)
  mean_x <- mean(x)
  sd_x <- sd(x)
  error <- qt(0.975, df = n - 1) * sd_x / sqrt(n)
  lower <- mean_x - error
  upper <- mean_x + error
  return(c(Mean = mean_x, LowerCI = lower, UpperCI = upper))
}

# Initialize results dataframe
results <- data.frame(
  Model = character(),
  MeanAccuracy = numeric(),
  LowerCI = numeric(),
  UpperCI = numeric(),
  stringsAsFactors = FALSE
)

# Calculate accuracy and confidence intervals for each model
for (model_name in model_names) {
  acc <- resamps$values[[paste0(model_name, "~Accuracy")]]
  stats <- getMeanCI(acc)
  results <- rbind(results, data.frame(
    Model = model_name,
    MeanAccuracy = round(stats["Mean"], 4),
    LowerCI = round(stats["LowerCI"], 4),
    UpperCI = round(stats["UpperCI"], 4)
  ))
}

Compiling Final Results

We compile the cross-validation and test set performance metrics into a comparison table.

# -------------------------
# Extract Performance Metrics
# -------------------------

# Function to extract CV metrics
extract_cv_metrics <- function(model, model_name, tune_param) {
  # Get the best tuning parameters
  best_params <- model$bestTune
  
  # Filter results for the best parameters
  best_results <- model$results
  for (param in names(best_params)) {
    best_results <- best_results[best_results[[param]] == best_params[[param]], ]
  }
  
  # Extract metrics
  accuracy <- best_results$Accuracy
  kappa <- best_results$Kappa
  accuracy_sd <- best_results$AccuracySD
  kappa_sd <- best_results$KappaSD
  
  # Compile into a data frame
  cv_metrics <- data.frame(
    Model = model_name,
    Metric = c("Accuracy (CV)", "Kappa (CV)"),
    Mean = c(accuracy, kappa),
    SD = c(accuracy_sd, kappa_sd),
    stringsAsFactors = FALSE
  )
  return(cv_metrics)
}

# Extract CV metrics for Random Forest
cv_metrics_rf <- extract_cv_metrics(model_rf_tuned, "Random Forest", "mtry")

# Extract CV metrics for Multinomial Logistic Regression
cv_metrics_multinom <- extract_cv_metrics(model_multinom_tuned, "Multinomial Logistic Regression", "decay")

# Function to extract Test metrics
extract_test_metrics <- function(conf_matrix, model_name) {
  accuracy <- conf_matrix$overall['Accuracy']
  kappa <- conf_matrix$overall['Kappa']
  
  test_metrics <- data.frame(
    Model = model_name,
    Metric = c("Accuracy (Test)", "Kappa (Test)"),
    Mean = as.numeric(c(accuracy, kappa)),
    SD = c(NA, NA),
    stringsAsFactors = FALSE
  )
  return(test_metrics)
}

# Extract Test metrics for Random Forest
test_metrics_rf <- extract_test_metrics(rf_confusion, "Random Forest")

# Extract Test metrics for Multinomial Logistic Regression
test_metrics_multinom <- extract_test_metrics(multinom_confusion, "Multinomial Logistic Regression")

# -------------------------
# Compile Final Results
# -------------------------
