# Import necessary libraries
import numpy as np
import pandas as pd
import matplotlib.pyplot as plt
from tensorflow.keras.models import Sequential
from tensorflow.keras.layers import LSTM, Dense

# Set random seed for reproducibility
np.random.seed(123)

# Simulate time series data
time_steps = 300
Time = np.arange(1, time_steps + 1)
Incidence = np.sin(Time * 0.1) + np.random.normal(scale=0.1, size=time_steps)
data = pd.DataFrame({'Time': Time, 'Incidence': Incidence})

# Plot the data
plt.figure(figsize=(10, 6))
plt.plot(data['Time'], data['Incidence'])
plt.xlabel('Time')
plt.ylabel('Incidence')
plt.title('Simulated Time Series Data')
plt.show()

# Normalize the data
max_incidence = data['Incidence'].max()
min_incidence = data['Incidence'].min()
data['Incidence_scaled'] = (data['Incidence'] - min_incidence) / (max_incidence - min_incidence)

# Prepare sequences
def create_sequences(data, seq_length):
    X = []
    y = []
    incidence_scaled = data['Incidence_scaled'].values
    for i in range(len(incidence_scaled) - seq_length):
        X.append(incidence_scaled[i:(i + seq_length)])
        y.append(incidence_scaled[i + seq_length])
    X = np.array(X)
    y = np.array(y)
    return X.reshape((X.shape[0], X.shape[1], 1)), y

seq_length = 10
X, y = create_sequences(data, seq_length)

# Split into training and testing sets
train_size = int(len(y) * 0.8)
X_train = X[:train_size]
y_train = y[:train_size]
X_test = X[train_size:]
y_test = y[train_size:]

# Build the model
model = Sequential()
model.add(LSTM(units=50, input_shape=(seq_length, 1)))
model.add(Dense(1))

# Compile the model
model.compile(loss='mean_squared_error', optimizer='adam')

# View model summary
model.summary()

# Predict on test data
predictions = model.predict(X_test)

# Rescale predictions
predictions_rescaled = predictions * (max_incidence - min_incidence) + min_incidence
y_test_rescaled = y_test * (max_incidence - min_incidence) + min_incidence

# Plot actual vs predicted
test_time = data['Time'][train_size + seq_length:]

df_results = pd.DataFrame({
    'Time': test_time.values,
    'Actual': y_test_rescaled,
    'Predicted': predictions_rescaled.flatten()
})

plt.figure(figsize=(10, 6))
plt.plot(df_results['Time'], df_results['Actual'], label='Actual', color='blue')
plt.plot(df_results['Time'], df_results['Predicted'], label='Predicted', color='red')
plt.xlabel('Time')
plt.ylabel('Incidence')
plt.title('Disease Incidence Forecasting')
plt.legend()
